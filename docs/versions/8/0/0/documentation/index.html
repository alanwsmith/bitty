this.api methods

Overview

Details

getStorageOr(key, alt)
  • Pulls an item out of localStorage and parses it as JSON
  • If there's not an item in localStorage with the given name the alternate value is returned
Examples
Get existing value and parse it as JSON
JavaScript
<script>
window.Class475EF = class {
  #storageKey = "key-475EF";

  bittyReady() {
    this.api.setStorage(
      this.#storageKey,
      { status: "ok" },
    );
    this.api.trigger("signal_475EF");
  }

  signal_475EF(_, el) {
    const data = this.api.getStorageOr(
      this.#storageKey,
      { status: "bug" },
    );
    el.innerHTML = data.status;
  }
};
</script>
HTML
<bitty-8-0 data-connect="Class475EF"></bitty-8-0>

<div class="test" data-receive="signal_475EF">bug</div>

sleep(ms)
Examples
Use await this.api.sleep(ms) to sleep for a given number the given number of milliseconds
JavaScript
<script>
window.Class7CEE5 = class {
  bittyReady() {
    this.api.trigger("signal_7CEE5");
  }

  async signal_7CEE5(_, el) {
    const t0 = performance.now();
    await this.api.sleep(50);
    const t1 = performance.now();
    if ((t1 - t0) > 45) {
      el.innerHTML = "ok";
    }
  }
};
</script>

window.ClassDDC6C = class {

    this.#delta = t1 - t0;
    this.api.trigger("signal_DDC6C");
