data-* attributes

Overview

  • The HTML side of the bitty API consists of these four attributes.
  • Attributes can be added to HTML elements to send and receive signals through JavaScript methods.

Items

data-connect
Description
  • `data-connect` is an optional attribute that can be added to bitty-8-0 tags.
  • If a `data-connect` attribute exists bitty will attempt to load the module defined by the attribute's value.
  • If a bitty-8-0 tag doesn't have a `data-connect` attribute it will attempt to load a module class from `window.BittyClass` as a fallback.
Examples
Load a default class from `window.BittyClass` if no `data-connect` attribute exists on a `bitty-8-0` tag. - [0010-default-bitty-class-on-window]
HTML
<bitty-8-0></bitty-8-0>
<div class="test" data-receive="signal_DA553">bug</div>
JavaScript
<script>
window.BittyClass = class {
  bittyReady() {
    this.trigger("signal_DA553");
  }

  signal_DA553(_, el) {
    el.innerHTML = "ok";
  }
};
</script>

window.Class9CE5F = class {

    testSender.dispatchEvent(testEvent);
  }

  signal_9CE5F(_, el) {
    el.innerHTML = "ok";
  }
};

// This test uses a custom event instead of
// something like mouseover to facilitate
// automated test output.

  constructor() {
    super("bittylocallistenertest", { bubbles: true });
  }
}
</script>

HTML

  class="test" 
  data-receive="signal_CBE64 signal_CBE64_2 signal_CBE64_3"

<script>
window.ClassCBE64 = class {
  #updates = [];

  bittyReady() {
    this.trigger(
      `signal_CBE64 signal_CBE64_2 signal_CBE64_3`,
    );
  }

  signal_CBE64(_, el) {
    this.#updates.push("bug");
    el.innerHTML = this.#updates[0];
  }

  signal_CBE64_2(_, el) {
    this.#updates.push("bug");
    el.innerHTML = this.#updates[1];
  }

  signal_CBE64_3(_, el) {
    if (this.#updates.length === 2) {
      el.innerHTML = "ok";
    }
  }
};
</script>

loadJSON(key, fallback)
Description

Load a JSON object into the `this.json[key]` collection.

Examples
Load JSON from localStorage into `this.json[key]` - [0010-json-exists-in-local-storage]
HTML
<bitty-8-0 data-connect="Class4B0E0"></bitty-8-0>
<div class="test" data-receive="verify_signal_4B0E0">bug</div>
JavaScript
<script>
window.Class4B0E0 = class {
  bittyReady() {
    this.trigger("given_signal_4B0E0");
  }

  given_signal_4B0E0(_, __) {
    const jsonAsString = `{ "data": { "status": "ok"} }`;
    localStorage.setItem("test_signal_4B0E0", jsonAsString);
    this.trigger("verify_signal_4B0E0");
  }

  verify_signal_4B0E0(_, el) {
    const result = this.loadJSON("test_signal_4B0E0");
    if (result.ok === true) {
      el.innerHTML = this.json["test_signal_4B0E0"].status;
    }
  }
};
</script>
